#include <LiquidCrystal.h>
#include <DHT.h>
const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 7;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);
String quality = "";
int gasLevel  = 0; 
#define sensor    A0
#define DHTPIN 2
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE); 
void setup() {
Serial.begin(9600);
dht.begin();
lcd.begin(20, 4);
lcd.print("Made By Basil");
delay(1000);
lcd.clear();
}

void loop() {
  delay(680);
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  gasLevel = analogRead(sensor);
  if(gasLevel<375){
    quality = "  GOOD!";
  }
  else if (gasLevel >376 && gasLevel<500){
    quality =  "  Poor!";
  }

 else if (gasLevel >501 && gasLevel<1250){
    quality =  "  TOXIC!";
     analogWrite(6, 550);
    delay(10000);
    analogWrite(6, 0);
  }
   lcd.setCursor(0,0);
   lcd.print("Air Quality : ");
    lcd.setCursor(13,0);
    lcd.print(quality);
    lcd.setCursor(7,1);
    lcd.print(t);
    lcd.setCursor(0,1);
    lcd.print("Temp : ");
    lcd.setCursor(12,2);
    lcd.print(h);
    lcd.setCursor(0,2);
    lcd.print("Humidity : ");
    delay(45);
    lcd.clear();
     lcd.setCursor(0,0);
   lcd.print("Air Quality : ");
    lcd.setCursor(13,0);
    lcd.print(quality);
     lcd.setCursor(7,1);
    lcd.print(t);
    lcd.setCursor(0,1);
    lcd.print("Temp : ");
    lcd.setCursor(12,2);
    lcd.print(h);
    lcd.setCursor(0,2);
    lcd.print("Humidity : ");
}
